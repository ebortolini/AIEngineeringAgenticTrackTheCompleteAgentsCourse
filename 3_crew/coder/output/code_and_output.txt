# Calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ...

# Initialize variables
n_terms = 10000
series_sum = 0.0

# Calculate the series
for i in range(n_terms):
    term = ((-1) ** i) / (2 * i + 1)
    series_sum += term

# Multiply the total by 4
result = series_sum * 4

# Output the result
print(result)
